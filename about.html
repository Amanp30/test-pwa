function checkAgeLessThan18(age) {
    return age <= 18 ? 'Underage' : '18 or older';
}

// Example Usage
console.log(checkAgeLessThan18(17));  // Underage
console.log(checkAgeLessThan18(18));  // 18 or older
console.log(checkAgeLessThan18(25));  // 18 or older


function checkVoterEligibility(age, citizenship) {
    return age >= 18 && citizenship === 'Indian'
        ? 'Eligible to Vote'
        : 'Not Eligible to Vote';
}

// Example Usage
console.log(checkVoterEligibility(20, 'Indian'));   // Eligible to Vote
console.log(checkVoterEligibility(17, 'Indian'));   // Not Eligible to Vote
console.log(checkVoterEligibility(20, 'American')); // Not Eligible to Vote
